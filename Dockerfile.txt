# Sử dụng Nginx bản nhẹ
FROM nginx:alpine

# Xóa các file mặc định cũ của Nginx để tránh xung đột
RUN rm /usr/share/nginx/html/*

# Copy toàn bộ file từ máy bạn vào container
COPY . /usr/share/nginx/html

# Ép Nginx phải dùng MAPP.html làm trang chủ (không cần đổi tên file gốc)
RUN printf "server {\n    listen 80;\n    location / {\n        root /usr/share/nginx/html;\n        index MAPP.html;\n    }\n}" > /etc/nginx/conf.d/default.conf

# Mở cổng 80
EXPOSE 80
